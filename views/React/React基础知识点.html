                                # 在React中对数组的处理

                                需要注意的几点

                                1. 在rander函数中 一旦需要进行数据处理的话需要在 { } 进行。
                                2. 循环数组的时候不要忘了加上key值，千万别重复。
                                constructor(props){
    super(props) //调用父类的构造函数，固定写法
    this.state={
        inputValue:"",
        list:['你好','世界']  
    }
}

render(){
    return  (
        <Fragment>
            <div>
                <input value={this.state.inputValue} onChange={this.inputChange.bind(this)} />
                <button> 增加字段 </button>
            </div>
            <ul>
                {
                    this.state.list.map((item,index)=>{
                        return <li key={index+item}>{item}</li>
                    })
                }
            </ul>  
        </Fragment>
    )
}

                                # 子组件调用父组件方法

                                如果子组件要调用父组件方法，其实和传递数据差不多，只要在组件调用时，把方法传递给子组件就可以了,记得这里也要进行this的绑定，如果不绑定子组件是没办法找到这个父组件的方法的。

                                // App.js
<ul>
    {
        this.state.list.map((item,index)=>{
            return (
                <ComponentChild 
                key={index+item}  
                content={item}
                index={index}
                handleClick={this.handleClick.bind(this)}
                />
            )
        })
    }
</ul>  

// ComponentChild.js
handleClick(){
    this.props.handleClick(this.props.index)
}

                                # React高级-生命周期

                                由图可知 React 声明周期的四个大阶段

                                1. Initialization:初始化阶段
                                2. Mounting: 挂在阶段
                                3. Updation: 更新阶段
                                4. Unmounting: 销毁阶段

                                直接上代码

                                import React, { Component, Fragment } from 'react';
class Life extends Component {
    constructor(props) {
        super(props);
        this.state = {};
    }

    componentWillMount() {
        // 1模板渲染前
        console.log("componentWillMount : 在组件即将被挂载到页面的时刻执行")
    }

    componentDidMount() {
        // 3模板渲染后
        console.log('componentDidMount----组件挂载完成的时刻执行')
    }
    render() {
        // 2模板渲染中
        return (
            <Fragment>

            </Fragment>
        );
    }

    // componentWillMount----组件将要挂载到页面的时刻执行
    // render----开始挂载渲染
    // componentDidMount----组件挂载完成的时刻执行

    shouldComponentUpdate(nextProps,nextState) {
        // nextProps:变化后的属性;
        // nextState:变化后的状态;
        // 判断模块需不需要渲染
        // 它要求返回一个布尔类型的结果，必须有返回值，这里就直接返回一个true了
        console.log('shouldComponentUpdate---组件发生改变前执行')

        // if (nextProps.content !== this.props.content) {
        //     return true
        // } else {
        //     return false
        // }
        return true
    }

    //shouldComponentUpdate 返回true才会被执行。
    componentWillUpdate() {
        // 如果需要渲染就执行该方法
        console.log('componentWillUpdate---组件更新前，shouldComponentUpdate函数之后执行')
    }

    componentDidUpdate() {
        // 模块渲染结束
        console.log('componentDidUpdate----组件更新之后执行')
    }

    // 1-shouldComponentUpdate---组件发生改变前执行
    // 2-componentWillUpdate---组件更新前，shouldComponentUpdate函数之后执行
    // 3-render----开始挂载渲染
    // 4-componentDidUpdate----组件更新之后执行

    componentWillReceiveProps() {
        // 模块将接受新的数据
        console.log('componentWillReceiveProps')
    }

    //当组件从页面中删除的时候执行
    componentWillUnmount() {
        console.log('child - componentWillUnmount')
    }
}

export default Life;

                                # npm install -save 和 -save-dev分不清

                                • npm install xxx: 安装项目到项目目录下，不会将模块依赖写入devDependencies或dependencies。
                                • npm install -g xxx: -g的意思是将模块安装到全局，具体安装到磁盘哪个位置，要看 npm cinfig prefix的位置。
                                • npm install -save xxx：-save的意思是将模块安装到项目目录下，并在package文件的dependencies节点写入依赖。
                                • npm install -save-dev xxx：-save-dev的意思是将模块安装到项目目录下，并在package文件的devDependencies节点写入依赖。
