      # React-Router的使用

      安装 react-router-dom

      npm install --save react-router-dom

      # 编写一个最简单的路由程序

      import React from "react";
import { BrowserRouter as Router, Route, Link } from "react-router-dom";

function Index() {
  return <h2>Index</h2>;
}

function List() {
  return <h2>List</h2>;
}

function AppRouter() {
  return (
    <Router>
        <ul>
            <li> <Link to="/">首页</Link> </li>
            <li><Link to="/list/">列表</Link> </li>
        </ul>
        <Route path="/" exact component={Index} />
        <Route path="/list/" component={List} />
    </Router>
  );
}
export default AppRouter;

      需要了解的事项

      1. Link 类似 HTML 中的a标签，to中地址是需要跳转的地址
      2. 使用的时候最外层需要用 Router 标签包裹
      3. Route 是视图显示区域 path地址对应着 Link 的 to 地址 component内含有需要显示的视图组件
      4. exact 精准匹配 表示路径信息要完全匹配成功，才可以实现跳转，匹配一部分是不行的。

      # 动态传值

      import React, { Component } from 'react';
import { Link } from "react-router-dom";

class Index extends Component {
    constructor(props) {
        super(props);
        this.state = { 
            list:[
                {uid:123,title:"值-1''},
                {uid:456,title:'值-2'}
            ]
         }
    }
    render() { 
        return ( 
            <ul>
                {
                    this.state.list.map((item,index)=>{
                        return (
                            <li key={index}>
                               <Link to={'/list/'+item.uid}> {item.title}</Link> 
                            </li>
                        )
                    })
                }
            </ul>
        )
    }
}
 
export default Index;

      # ReactRouter重定向-Redirect

      重定向方式有两种

      重定向和跳转有一个重要的区别，就是跳转式可以用浏览器的回退按钮返回上一级的，而重定向是不可以的。

      1. 标签式重定向:就是利用标签来进行重定向，业务逻辑不复杂时建议使用这种。
      2. 编程式重定向:这种是利用编程的方式，一般用于业务逻辑当中，比如登录成功挑战到会员中心页面。
      // 标签重定向
function Home() {
  return <h2>Home</h2>;
}

 <Route path="/home/" component={Home} />
 <Redirect to="/home/" />
