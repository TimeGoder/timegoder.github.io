# MySQL的基础使用

我买了本mysql应知应会 用来学习前期用命令行 后期用 navacat 了 主要是记录我自己所想知道的一些情况

这是 student_info库中的 student_grade 表的内容

话不多说 直接上语句吧 这是我第一天所学的

-- SELECT sex,s_match,s_chinese,s_english FROM student_grade;

-- SELECT * FROM student_grade LIMIT 2,5;
-- 从第2（起始0）行开始取五个

-- SELECT student_grade.`name` FROM student_grade;
-- 只从student_grade中连接

-- SELECT * FROM student_grade ORDER BY age;
-- ORDER BY 子句可以指定顺序输出 

-- SELECT * FROM student_grade ORDER BY age;
-- SELECT * FROM student_grade ORDER BY age DESC;
-- SELECT * FROM student_grade ORDER BY age DESC,s_match;
-- SELECT * FROM student_grade ORDER BY age DESC LIMIT 1;
-- DESC 为降序排序 默认是升序

-- SELECT * FROM student_grade WHERE age = 22 ORDER BY s_chinese DESC LIMIT 1;
-- WHERE语句位于表名之后, ORDER BY 必须位于WHERE子句之后, LIMIT位于 ORDER BY之后

-- SELECT * FROM student_grade WHERE age BETWEEN 22 AND 25 ORDER BY s_match DESC LIMIT  5;
-- 查询符合age在22到25之间的所有数据，并且依照 s_match 从高到底排序输出5条记录


-- SELECT * FROM student_grade WHERE age>22 AND age<=25;
-- 输出 22 25 之间的语句

-- SELECT * FROM student_grade WHERE ( id = 5 OR id = 10 ) AND age = 23;
-- 计算顺序 括号 > AND > or; 

-- SELECT * FROM student_grade WHERE id IN (1,5) ORDER BY age DESC LIMIT 2;
-- 匹配 id 1和5的数据按照age从大到小输出两条

-- SELECT * FROM student_grade WHERE id NOT IN (1,5) ORDER BY age DESC LIMIT 10;
-- 匹配 不是id 1和5的数据按照age从大到小输出10条

-- SELECT * FROM student_grade WHERE s_motto LIKE "%冲啊%";
-- 查询 s_motto 含有 冲啊 的内容
  
-- SELECT CONCAT(name,"今年",age,"岁了，","他的座右铭是",s_motto) AS motto FROM student_grade;
-- 使用concat 可以连接多个字符

--  SELECT *,(s_chinese + s_match + s_english) AS total_score,NOW() as query_date FROM student_grade ORDER BY total_score DESC;
